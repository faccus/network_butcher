751  template <typename GraphType>
752  void
753  Constrained_Block_Graph_Builder<GraphType>::apply_operation_weights(Block_Graph_Type &new_graph) const
754  {
755  using namespace network_butcher::parameters;
756 
757  // If we can import the weight with an importer, we return. Otherwise, we read the weights from the original graph
758  if (apply_weights_from_importer(new_graph))
759  return;
760 
761  auto const &nodes = new_graph.get_nodes();
762  auto const &graph = this->original_graph;
763  auto const &mode = block_graph_generation_params.block_graph_mode;
764 
765  auto const process_node = [&new_graph, &graph, &mode](auto const &node) {
766  auto const first = node.get_id();
767 
768  // Look for the nodes of the original graph that are
769  // represented by the input node (in the linearized
770  // graph)
771  auto const &inputs = *node.content.second;
772 
773  // The device id of the input node (=0 starting device, >0 other device)
774  auto const in_device_id = node.content.first;
775 
776  for (auto const &second : new_graph.get_output_nodes(first))
777  {
778  auto const &out_node = new_graph[second];
779 
780  Edge_Type const edge = std::make_pair(first, second);
781  // The device id of the output node (=0 starting device, >0 other device)
782  auto const out_device_id = out_node.content.first;
783 
784  // Look for the nodes of the original graph that are represented by the output node (in the
785  // linearized graph)
786  auto const &outputs = *out_node.content.second;
787 
788  Time_Type weight_cost = 0.;
789 
790  // 1-1 correspondence
791  if (outputs.size() == 1 && inputs.size() == 1)
792  {
793  auto const &input = *inputs.begin();
794  auto const &output = *outputs.begin();
795 
796  auto const tmp_edge = std::make_pair(input, output);
797 
798  weight_cost = graph.get_weight(out_device_id, tmp_edge);
799  }
800  // (2+)-1 correspondence
801  else if (outputs.size() == 1)
802  {
803  auto const &output = *outputs.begin();
804  auto const &inputs_of_output = graph.get_input_nodes(output);
805 
806  weight_cost = graph.get_weight(out_device_id, std::make_pair(*inputs_of_output.cbegin(), output));
807  }
808  // 1-(2+) correspondence
809  else if (inputs.size() == 1)
810  {
811  auto const &input = *inputs.begin();
812  auto const &interface_outputs = graph.get_output_nodes(input);
813 
814  for (auto const &output : interface_outputs)
815  weight_cost += graph.get_weight(out_device_id, std::make_pair(input, output));
816 
817  // Compute the total weight associated to the internal edges
818  for (auto const &internal_input : outputs)
819  {
820  for (auto &internal_output : graph.get_output_nodes(internal_input))
821  {
822  if (outputs.find(internal_output) != outputs.cend())
823  {
824  weight_cost +=
825  graph.get_weight(out_device_id, std::make_pair(internal_input, internal_output));
826  }
827  }
828  }
829  }
830  // (2+)-(2+). In this case, there are two possibilities: either the block graph mode is classic
831  // (and the program should trow) or the block graph mode is input/output. In the latter case, we
832  // should consider the transmission of the "frontier" nodes of inputs and the overall execution
833  // cost for the outputs
834  else
835  {
836  // In classic mode, every edge can have at most one 2+ node.
837  if (mode == Block_Graph_Generation_Mode::classic)
838  {
839  throw std::logic_error("The edge (" + std::to_string(edge.first) + ", " +
840  std::to_string(edge.second) + ") has both multiple inputs and outputs!");
841  }
842  // In input and output mode, every edge can have up to two 2+ nodes.
843  else
844  {
845  bool set = false;
846 
847  // Compute the total weight associated to the internal edges
848  for (auto const &internal_input : outputs)
849  {
850  for (auto &internal_output : graph.get_output_nodes(internal_input))
851  {
852  if (outputs.find(internal_output) != outputs.cend())
853  {
854  weight_cost +=
855  graph.get_weight(out_device_id, std::make_pair(internal_input, internal_output));
856  set = true;
857  }
858  }
859  }
860 
861  if (!set)
862  {
863  std::stringstream stt;
864  stt << "Missing weight in block graph generation!" << std::endl;
865 
866  throw std::logic_error(stt.str());
867  }
868  }
869  }
870 
871  new_graph.set_weight(edge, weight_cost);
872  }
873  };
874 
875 #if PARALLEL_TBB
876  std::for_each(nodes.cbegin(), nodes.cend(), process_node);
877 #else
878 # pragma omp parallel default(none) shared(nodes, new_graph, graph, mode, process_node)
879  {
880 # pragma omp for
881  for (const auto &node : nodes)
882  {
883  process_node(node);
884  }
885  }
886 #endif
887  }

890  template <typename GraphType>
891  void
892  Constrained_Block_Graph_Builder<GraphType>::apply_transmission_weights(Block_Graph_Type &new_graph) const
893  {
894  using namespace network_butcher::parameters;
895 
896  auto const &nodes = new_graph.get_nodes();
897  auto const &graph = this->original_graph;
898  auto const &mode = block_graph_generation_params.block_graph_mode;
899  auto const &ts_weights = transmission_weights;
900 
901  auto const process_node = [&new_graph, &graph, &mode, &ts_weights](auto const &node) {
902  auto const &inputs = *node.content.second;
903  auto const first = node.get_id();
904 
905  auto const in_device_id = node.content.first;
906 
907  for (auto const &second : new_graph.get_output_nodes(first))
908  {
909  auto const &out_node = new_graph[second];
910 
911  Edge_Type const edge = std::make_pair(first, second);
912  // The device id of the output node (=0 starting device, >0 other device)
913  auto const out_device_id = out_node.content.first;
914 
915  // Look for the nodes of the original graph that are represented by the output node (in the
916  // linearized graph)
917  auto const &outputs = *out_node.content.second;
918 
919  Time_Type final_cost = 0.;
920 
921  // 1-1 correspondence
922  if (outputs.size() == 1 && inputs.size() == 1)
923  {
924  auto const &input = *inputs.begin();
925  auto const &output = *outputs.begin();
926 
927  auto const tmp_edge = std::make_pair(input, output);
928 
929  final_cost = ts_weights(tmp_edge, in_device_id, out_device_id);
930  }
931  // (2+)-1 correspondence. The idea is that the input nodes must transmit to the output node the
932  // different values. Thus, the transmission cost is paid several times.
933  else if (outputs.size() == 1)
934  {
935  auto const &output = *outputs.begin();
936  // The inputs on the original graph of the output node have to
937  // transmit their values to the output node
938  for (auto const &input : graph.get_input_nodes(output))
939  {
940  final_cost += ts_weights(std::make_pair(input, output), in_device_id, out_device_id);
941  }
942  }
943  // 1-(2+). In this case, the input is sent to the device of the output nodes a single time.
944  // Thus, this transmission cost is taken into account only once.
945  else if (inputs.size() == 1)
946  {
947  auto const &input = *inputs.begin();
948  auto const &comm_outputs = graph.get_output_nodes(input);
949 
950  final_cost += ts_weights(std::make_pair(input, *comm_outputs.crbegin()), in_device_id, out_device_id);
951  }
952  // (2+)-(2+). In this case, there are two possibilities: either the block graph mode is classic
953  // (and the program should trow) or the block graph mode is input/output. In the latter case, we
954  // should consider the transmission of the "frontier" nodes of inputs and the overall execution
955  // cost for the outputs
956  else
957  {
958  // In classic mode, every edge can have at most one 2+ node.
959  if (mode == Block_Graph_Generation_Mode::classic)
960  {
961  throw std::logic_error("The edge (" + std::to_string(edge.first) + ", " +
962  std::to_string(edge.second) + ") has both multiple inputs and outputs!");
963  }
964  // In input and output mode, every edge can have up to two 2+ nodes.
965  else
966  {
967  // This is the collection of the input nodes of every node contained in outputs
968  std::set<Node_Id_Type> output_node_inputs;
969 
970  for (auto const &node_id : outputs)
971  {
972  auto const &tmp_nodes = graph.get_input_nodes(node_id);
973  output_node_inputs.insert(tmp_nodes.cbegin(), tmp_nodes.cend());
974  }
975 
976  // This is the collection of nodes in inputs whose output tensors are fed to outputs
977  std::vector<Node_Id_Type> frontier_input(std::max(inputs.size(), output_node_inputs.size()));
978  auto const close_frontier = std::set_intersection(output_node_inputs.cbegin(),
979  output_node_inputs.cend(),
980  inputs.cbegin(),
981  inputs.cend(),
982  frontier_input.begin());
983 
984  // We have to consider the transmission cost for every node in the frontier_input
985  for (auto input_it = frontier_input.cbegin(); input_it != close_frontier; ++input_it)
986  {
987  final_cost += ts_weights(std::make_pair(*input_it, *graph.get_output_nodes(*input_it).crbegin()),
988  in_device_id,
989  out_device_id);
990  }
991  }
992  }
993 
994  if (new_graph.check_weight(edge))
995  final_cost += new_graph.get_weight(edge);
996 
997  new_graph.set_weight(edge, final_cost);
998  }
999  };
1000 
1001 #if PARALLEL_TBB
1002  std::for_each(std::execution::par, nodes.cbegin(), nodes.cend(), process_node);
1003 #else
1004 # pragma omp parallel default(none) shared(nodes, graph, new_graph, transmission_weights, mode, process_node)
1005  {
1006 # pragma omp for
1007  for (const auto &node : nodes)
1008  {
1009  process_node(node);
1010  }
1011  }
1012 #endif
1013  }
