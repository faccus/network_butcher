114  auto
115  KFinder_Eppstein<Graph_type, Only_Distance, t_Weighted_Graph_Complete_Type>::construct_h_out(
116  std::vector<Node_Id_Type> const &successors,
117  Internal_Weight_Collection_Type const &sidetrack_distances) const -> H_out_collection
118  {
119  auto const &graph = Parent_Type::graph;
120 
121  H_out_collection h_out_collection;
122  h_out_collection.reserve(graph.size());
123 
124  for (auto const &tail_node : graph)
125  {
126  auto const &tail = tail_node.get_id();
127  auto const &tail_successor = successors[tail];
128 
129  if (tail_successor == std::numeric_limits<Node_Id_Type>::max())
130  continue;
131 
132  // The output neighbors of the current node
133  auto const &head_nodes = graph.get_output_nodes(tail);
134  std::vector<Edge_Info> sidetrack_edges;
135  sidetrack_edges.reserve(sidetrack_distances[tail].size());
136 
137  // Loop through the output neighbors of the current node
138  for (auto const &head : head_nodes)
139  {
140  auto [begin, end] = sidetrack_distances[tail].equal_range(head);
141 
142  for (; begin != end && begin->first == head; ++begin)
143  {
144  sidetrack_edges.emplace_back(std::make_pair(tail, head), begin->second);
145  }
146  }
147 
148  // Add to the collection of H_outs
149  h_out_collection.emplace_hint(h_out_collection.end(),
150  tail,
151  typename H_out_collection::mapped_type(std::move(sidetrack_edges)));
152  }
153 
154  return h_out_collection;
155  }

159  auto
160  KFinder_Eppstein<Graph_type, Only_Distance, t_Weighted_Graph_Complete_Type>::construct_h_g(
161  const H_out_collection &h_out_collection,
162  const std::vector<Node_Id_Type> &successors) const -> H_g_collection
163  {
164  auto const &num_nodes = graph.size();
165 
166  H_g_collection h_g_collection;
167  h_g_collection.reserve(graph.size());
168 
169  // sp_dependencies contains the predecessors of every node in the shortest path.
170  std::vector<std::set<Node_Id_Type>> sp_dependencies;
171  sp_dependencies.resize(num_nodes);
172 
173  // At worst, since each node is contained in up to a single sp_dependencies, the loop takes at most O(N*log(N))
174  for (auto const &node : graph) // O(N)
175  {
176  auto const &node_id = node.get_id();
177  auto const &successor = successors[node_id];
178 
179  if (successor == std::numeric_limits<Node_Id_Type>::max())
180  continue;
181 
182  if (sink != node_id)
183  sp_dependencies[successor].insert(node_id); // O(log(N))
184  }
185 
186  // The actual generation of the H_g should now start from the sink node
187  auto h_out_iterator = h_out_collection.find(sink); // O(1), it always exists
188 
189  // Prepare the last H_g
190  h_g_collection.emplace(sink, typename H_g_collection::mapped_type(&h_out_iterator->second)); // O(1)
191 
192  // Now, we have to find the nodes whose successor is the sink itself
193  std::queue<Node_Id_Type> queue;
194  for (auto const &node : graph) // O(N)
195  {
196  auto const &id = node.get_id();
197  if (successors[id] == sink && id != sink) // O(1)
198  queue.push(id);
199  }
200 
201  // Loop through the queue
202  while (!queue.empty()) // O(N)
203  {
204  auto const &front_element = queue.front();
205 
206  // Find the "new" nodes that must be added to the queue.
207  auto &deps = sp_dependencies[front_element]; // O(1)
208  auto const &successor_h_g = h_g_collection.find(successors[front_element])->second; // O(1)
209 
210  h_g_collection.emplace(front_element,
211  typename H_g_collection::mapped_type(&(h_out_collection.find(front_element)->second),
212  successor_h_g)); // O(log(N))
213 
214 
215  // Among the different iterations, this loop is performed at most N times. Moreover, every iteration of the
216  // loop will add a new node to the queue. This means that this loop doesn't change the worst case complexity
217  // of the overall method
218  for (auto &n : deps)
219  queue.push(n);
220 
221  queue.pop(); // O(1)
222  }
223 
224  return h_g_collection;
225  }
