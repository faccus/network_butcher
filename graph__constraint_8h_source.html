155  template <typename GraphType>
156  auto
157  Memory_Constraint<GraphType>::estimate_maximum_memory_usage(
158  const std::vector<network_butcher::parameters::Device> &devices,
159  const std::set<Node_Id_Type> &ids,
160  const std::vector<Memory_Type> &input_memory,
161  const std::vector<Memory_Type> &output_memory,
162  const std::vector<Memory_Type> &params_memory) const -> std::tuple<Memory_Type, Memory_Type>
163  {
164  Memory_Type result_memory = 0, fixed_memory = 0;
165 
166  // This variable will estimate the maximum number of branches detected. The resulting memory will be given by the
167  // space occupied by the parameters and the product between qty and the maximum sum between the input and output
168  // of each node. The final result will be an overestimation of the memory required by the group of nodes.
169  std::size_t qty = 1;
170 
171  // Memory required by the parameters
172  fixed_memory =
173  std::reduce(std::next(params_memory.begin(), *ids.cbegin()), std::next(params_memory.begin(), *ids.crbegin()));
174 
175  // Check if a branch is detected
176  if (graph.get_input_nodes(*ids.begin()).size() == 1)
177  {
178  auto const &father = *graph.get_input_nodes(*ids.begin()).begin();
179  qty = std::max(qty, graph.get_output_nodes(father).size());
180  }
181 
182  for (auto const &id : ids)
183  {
184  auto const &parents = graph.get_input_nodes(id);
185  auto const &children = graph.get_output_nodes(id);
186 
187  Memory_Type const &in_memory = input_memory[id];
188  Memory_Type const &out_memory = output_memory[id];
189 
190  // Compute the maximum memory required by any node
191  result_memory = std::max(result_memory, in_memory + out_memory);
192 
193  // Adjust qty if extra branches are detected
194  if (children.size() > parents.size())
195  qty = qty + children.size() - parents.size();
196  }
197 
198  return {result_memory * qty, fixed_memory};
199  }
