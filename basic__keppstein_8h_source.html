315  Basic_KEppstein<Graph_type, Only_Distance, t_Weighted_Graph_Complete_Type>::general_algo_eppstein(
316  std::size_t K,
317  Dijkstra_Result_Type const &dij_res,
318  Internal_Weight_Collection_Type const &sidetrack_distances,
319  H_g_collection &h_g,
320  H_out_collection &h_out,
321  const Basic_KEppstein::Callback_Function &callback_fun) const -> Output_Type
322  {
323  auto const &[successors, shortest_distance] = dij_res;
324 
325  // Start with the shortest path
326  std::vector<Implicit_Path_Info> res;
327  res.push_back(Implicit_Path_Info{.current_sidetrack = std::optional<Sidetrack>(),
328  .previous_sidetracks = nullptr,
329  .length = shortest_distance[root]});
330 
331  // Find the first sidetrack edge
332  typename H_g_collection::const_iterator h_g_it = h_g.find(root);
333 
334  if (h_g_it->second.empty())
335  {
336  if constexpr (Only_Distance)
337  {
338  return {shortest_distance[root]};
339  }
340  else
341  {
342  return {Shortest_path_finder::shortest_path_finder(Parent_Type::graph, dij_res, root, Parent_Type::sink)};
343  }
344  }
345 
346  res.reserve(K);
347 
348  // Collection of "final" implicit paths to be added to res
349  std::vector<Implicit_Path_Info> tmp_vect_queue;
350  tmp_vect_queue.reserve(K);
351  std::priority_queue<Implicit_Path_Info, std::vector<Implicit_Path_Info>, std::greater<>> Q(
352  std::greater<>(), std::move(tmp_vect_queue));
353 
354 
355  // First deviatory path
356  auto const &first_side_track = extract_first_sidetrack_edge(h_g_it);
357  Q.push(Implicit_Path_Info{.current_sidetrack = first_side_track,
358  .previous_sidetracks = nullptr,
359  .length = first_side_track.get_head_content().delta_weight + shortest_distance[root]});
360 
361 
362  std::size_t k = 2;
363 
364  // Loop through Q until either Q is empty, or the number of paths found is K
365  for (; k <= K && !Q.empty(); ++k)
366  {
367  res.emplace_back(Q.top());
368  Q.pop();
369 
370  auto const &SK = res.back();
371  auto const &current_sidetrack = SK.current_sidetrack.value();
372  auto const &[e_edge, e_weight] = current_sidetrack.get_head_content();
373 
374  // "Helper" function that can be called if needed
375  if (callback_fun != nullptr)
376  {
377  h_g_it = callback_fun(h_g, h_out, sidetrack_distances, successors, e_edge.second, graph);
378  }
379  else
380  {
381  h_g_it = h_g.find(e_edge.second);
382  }
383 
384  if (!h_g_it->second.empty())
385  {
386  // Extract the first sidetrack edge if it exists
387  auto const &f = extract_first_sidetrack_edge(h_g_it);
388  Q.push(Implicit_Path_Info{.current_sidetrack = f,
389  .previous_sidetracks = &res.back(),
390  .length = SK.length + f.get_head_content().delta_weight}); // O(log(K)
391  }
392  auto const &sidetrack_edges = get_alternatives(current_sidetrack);
393 
394  // O(1), there are up tp 3 elements in this collection
395  for (auto const &sidetrack_edge : sidetrack_edges)
396  {
397  Q.push(Implicit_Path_Info{.current_sidetrack = sidetrack_edge,
398  .previous_sidetracks = SK.previous_sidetracks,
399  .length = SK.length + sidetrack_edge.get_head_content().delta_weight -
400  e_weight}); // O(log(K))
401  }
402  }
403 
404  if constexpr (Only_Distance)
405  {
406  std::vector<Weight_Type> final_res;
407  final_res.reserve(res.size());
408 
409  for (auto const &path : res)
410  final_res.push_back(path.length);
411 
412  return final_res;
413  }
414  else
415  {
416  return helper_eppstein(dij_res, res);
417  }
418  }

463  auto
464  Basic_KEppstein<Graph_type, Only_Distance, t_Weighted_Graph_Complete_Type>::helper_eppstein(
465  Dijkstra_Result_Type const &dij_res,
466  std::vector<Implicit_Path_Info> const &epp_res) const -> std::vector<Path_Info>
467  {
468  auto const &root = Parent_Type::root;
469  auto const &sink = Parent_Type::sink;
470 
471  auto const &successors = dij_res.first;
472 
473  std::vector<Path_Info> res(epp_res.size());
474 
475  auto const go_shortest = [&successors, sink](Node_Id_Type node) {
476  std::vector<Node_Id_Type> final_steps;
477  while (node != sink)
478  {
479  final_steps.push_back(node);
480  node = successors[node];
481  }
482 
483  final_steps.push_back(sink);
484  return final_steps;
485  };
486 
487 
488  // Basically, we start from the specified node and go along the shortest path until we meet a sidetrack edge
489  // contained in the implicit path. In that case, we add the sidetrack edge and proceed along the "new" shortest
490  // path until either the "sink" node is reached or another sidetrack edge is met
491  auto const process_path =
492  [&go_shortest, &dij_res = dij_res, &res = res, &epp_res = epp_res, &root, &sink](std::size_t i) {
493  auto const &implicit_path = epp_res[i];
494 
495  auto &info = res[i];
496  auto const &sidetracks = implicit_path.compute_sidetracks();
497 
498  info.length = implicit_path.length;
499 
500  if (sidetracks.empty())
501  {
502  info.path = go_shortest(root);
503  }
504  else
505  {
506  auto sidetrack_edge_it = sidetracks.cbegin();
507 
508  auto [first, second] = (*sidetrack_edge_it)->edge;
509  std::size_t node_to_insert = root;
510 
511  while (node_to_insert != sink)
512  {
513  info.path.push_back(node_to_insert);
514  if (first == node_to_insert)
515  {
516  node_to_insert = second;
517  ++sidetrack_edge_it;
518 
519  if (sidetrack_edge_it == sidetracks.cend())
520  {
521  auto to_insert = go_shortest(node_to_insert);
522  info.path.insert(info.path.end(),
523  std::make_move_iterator(to_insert.begin()),
524  std::make_move_iterator(to_insert.end()));
525 
526  break;
527  }
528 
529  auto tmp = (*sidetrack_edge_it)->edge;
530  first = tmp.first;
531  second = tmp.second;
532  }
533  else
534  node_to_insert = dij_res.first[node_to_insert];
535  }
536  }
537  };
538 #if PARALLEL_TBB
539  // https://stackoverflow.com/a/63340360 .
540  // Tested views, but they are slower, so they are not used.
541 
542  std::vector<std::size_t> v(epp_res.size());
543  std::generate(v.begin(), v.end(), [n = 0]() mutable { return n++; });
544 
545  std::for_each(std::execution::par, v.begin(), v.end(), process_path);
546 #else
547 # pragma omp parallel default(none) shared(go_shortest, dij_res, res, epp_res, root, sink, process_path)
548  {
549 # pragma omp for
550  for (std::size_t i = 0; i < epp_res.size(); ++i)
551  {
552  process_path(i);
553  }
554  }
555 #endif
556 
557  return res;
558  }

561  template <typename Graph_type, bool Only_Distance, Valid_Weighted_Graph t_Weighted_Graph_Complete_Type>
562  auto
563  Basic_KEppstein<Graph_type, Only_Distance, t_Weighted_Graph_Complete_Type>::sidetrack_distances(
564  Dijkstra_Result_Type const &dij_res) const -> Internal_Weight_Collection_Type
565  {
566  Internal_Weight_Collection_Type res;
567  res.resize(graph.size());
568 
569  auto const &[successors, distances_from_sink] = dij_res;
570 
571  for (auto const &tail_node : graph) // The overall loop has worst case complexity O(E)
572  {
573  auto const &tail = tail_node.get_id();
574  if (distances_from_sink[tail] == std::numeric_limits<Weight_Type>::max())
575  continue;
576 
577  for (auto const &head : graph.get_output_nodes(tail))
578  {
579  if (distances_from_sink[head] == std::numeric_limits<Weight_Type>::max())
580  continue;
581 
582  auto const edge = std::make_pair(tail, head);
583  auto const &weights = graph.get_weight(edge);
584 
585  // If it is its successor...
586  if (successors[tail] == head)
587  {
588  // ...and it has more than one weight, then we have to consider the edge with the smallest weight as the
589  // edge in the shortest path tree, while the other ones will be sidetrack edges.
590  if (weights.size() > 1)
591  {
592  bool found = false;
593 
594  for (auto it = ++weights.cbegin(); it != weights.cend(); ++it)
595  {
596  auto const &weight = *it;
597 
598  res[tail].emplace_hint(res[tail].cend(),
599  head,
600  weight + distances_from_sink[head] - distances_from_sink[tail]); // O(1)}
601  }
602  }
603  }
604  else
605  {
606  for (auto const &weight : weights)
607  {
608  res[tail].emplace_hint(res[tail].cend(),
609  head,
610  weight + distances_from_sink[head] - distances_from_sink[tail]); // O(1)
611  }
612  }
613  }
614  }
615 
616  return res;
617  }
