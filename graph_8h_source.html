411  Graph<CNode<Content<T>>>::compute_dependencies()
412  {
413  // Reset the dependency vector.
414  neighbours = Neighbours_Type();
415  neighbours.resize(nodes.size());
416 
417  // Compute appearances of inputs/outputs for a node
418  std::unordered_map<std::string, Node_Id_Collection_Type> input_appearances;
419  std::unordered_map<std::string, Node_Id_Collection_Type> output_appearances;
420 
421  // Check which node has which input/output
422  for (auto const &node : nodes)
423  {
424  auto const &content = node.content;
425  for (auto &in : content.get_input())
426  input_appearances[in.first].insert(node.get_id());
427  for (auto &out : content.get_output())
428  output_appearances[out.first].insert(node.get_id());
429  }
430 
431  // Matched the input of a node to his outputs and viceversa
432  for (auto const &appearance : input_appearances)
433  {
434  auto const &neib = output_appearances[appearance.first];
435  for (auto node_id : appearance.second)
436  neighbours[node_id].first.insert(neib.cbegin(), neib.cend());
437  for (auto node_id : neib)
438  neighbours[node_id].second.insert(appearance.second.cbegin(), appearance.second.cend());
439  }
440  }
