408  Graph<CNode<Content<T>>>::compute_dependencies()
409  {
410  // Reset the dependency vector.
411  neighbours = Neighbours_Type();
412  neighbours.resize(nodes.size());
413 
414  // Compute appearances of inputs/outputs for a node
415  std::unordered_map<std::string, Node_Id_Collection_Type> input_appearances;
416  std::unordered_map<std::string, Node_Id_Collection_Type> output_appearances;
417 
418  // Check which node has which input/output
419  for (auto const &node : nodes)
420  {
421  auto const &content = node.content;
422  for (auto &in : content.get_input())
423  input_appearances[in.first].insert(node.get_id());
424  for (auto &out : content.get_output())
425  output_appearances[out.first].insert(node.get_id());
426  }
427 
428  // Matched the input of a node to his outputs and viceversa
429  for (auto const &appearance : input_appearances)
430  {
431  auto const &neib = output_appearances[appearance.first];
432  for (auto node_id : appearance.second)
433  neighbours[node_id].first.insert(neib.cbegin(), neib.cend());
434  for (auto node_id : neib)
435  neighbours[node_id].second.insert(appearance.second.cbegin(), appearance.second.cend());
436  }
437  }
